E - 8.5 (1).py

flowchart TD
A[Inicio] --> B[Leer entrada y separar tokens]
B --> C{¿Hay entrada?}
C -- No --> Z[Fin]
C -- Sí --> D[Leer n y cadena s]
D --> E[es_valido = True]
E --> F[Recorrer i=0..n-2]
F --> G{¿s[i]=='1' y s[i+1]=='1'?}
G -- Sí --> H[es_valido=False y salir]
G -- No --> I[Continuar]
I --> J{¿Termina ciclo?}
H --> K
J -- No --> F
J -- Sí --> K{¿es_valido?}
K -- No --> N[Imprimir No]
K -- Sí --> L[Revisar cada 0]
L --> M{¿Vecinos libres?}
M -- Sí --> O[es_valido=False]
M -- No --> P[Continuar]
P --> Q{¿Termina ciclo?}
O --> R{¿es_valido?}
Q -- No --> L
Q -- Sí --> R
R -- Sí --> S[Imprimir Yes]
R -- No --> N
S --> Z
N --> Z

F - 8.6 (1).py

flowchart TD
A[Inicio] --> B[Leer tokens]

C -- No --> Z[Fin]
C -- Sí --> D[p=0, heap=[]]
D --> E[Registrar consultas hasta '#']

J --> K[Repetir k veces]
K --> L[pop mínimo]
L --> M[Guardar qid]
M --> N[push con próximo tiempo]
N --> O{¿faltan iteraciones?}
O -- Sí --> K
O -- No --> P[Imprimir resultados]
P --> Z

G - 8.7 (1).py

flowchart TD
A[Inicio] --> B[Leer n]
B --> C[salud=0, tomadas=0, heap=[]]
C --> D[Para cada poción]
D --> E[Leer valor]
E --> F[salud += valor; tomadas += 1]
F --> G{¿valor < 0?}
G -- Sí --> H[push en heap]
G -- No --> I
H --> I{¿salud < 0?}
I -- Sí --> J[pop peor negativa]
J --> K[Revertir efecto y tomadas--]
K --> L
I -- No --> L{¿quedan pociones?}
L -- Sí --> D
L -- No --> M[Imprimir tomadas]
M --> Z[Fin]

M - 7.13 (1).py

O - 8.15 (1).py

flowchart TD
A[Inicio] --> B[Leer n por tokens]
B --> C{¿fin o n=0?}
C -- Sí --> Z[Imprimir salidas y fin]
C -- No --> D[Leer n números]
D --> E[heapify]
E --> F[costo=0]

G -- Sí --> H[pop dos mínimos]

P - 8.16 (1).py

flowchart TD
A[Inicio] --> B[Leer t casos]
B --> C[Para cada caso]
C --> D[Leer n y crear resultado]
D --> E[Heap de segmentos inicial]
E --> F[Para i=1..n]
F --> G[pop segmento más largo]
G --> H[calcular mid]
H --> I[resultado[mid]=i]
I --> J{¿hay subsegmento izq?}
J -- Sí --> K[push izq]
J -- No --> L
K --> L{¿hay subsegmento der?}
L -- Sí --> M[push der]
L -- No --> N
M --> N{¿sigue i?}
N -- Sí --> F
N -- No --> O[guardar respuesta caso]
O --> P{¿quedan casos?}
P -- Sí --> C
P -- No --> Q[imprimir todas]
Q --> Z[Fin]
